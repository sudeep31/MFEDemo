<template>
  <div class="container mt-4">
    <div class="alert alert-success">
      <h4>Vue MFE Loaded Successfully!</h4>
      <p class="mb-0">
        This is a <strong>Vue 3</strong> Micro Frontend running on port
        <code>3002</code>, loaded inside the Shell via
        <strong>Webpack 5 Module Federation</strong>.
      </p>
    </div>

    <!-- Session Stats Card — data injected by the Shell via provide/inject -->
    <div class="card mb-3 border-success">
      <div class="card-header bg-success text-white d-flex justify-content-between align-items-center">
        <span>Session Stats — passed from Shell</span>
        <span class="badge bg-light text-success fs-6">
          {{ stats.visitCount }} visit{{ stats.visitCount !== 1 ? 's' : '' }}
        </span>
      </div>
      <div class="card-body">
        <div class="row g-2 mb-3">
          <div class="col-6 col-md-3">
            <div class="p-2 bg-light rounded text-center">
              <div class="text-muted small">User</div>
              <div class="fw-semibold">{{ stats.userName || '—' }}</div>
            </div>
          </div>
          <div class="col-6 col-md-3">
            <div class="p-2 bg-light rounded text-center">
              <div class="text-muted small">Session Start</div>
              <div class="fw-semibold">{{ stats.sessionStartTime || '—' }}</div>
            </div>
          </div>
          <div class="col-6 col-md-3">
            <div class="p-2 bg-light rounded text-center">
              <div class="text-muted small">This MFE Visits</div>
              <div class="fw-semibold text-success">{{ stats.visitCount }}</div>
            </div>
          </div>
          <div class="col-6 col-md-3">
            <div class="p-2 bg-light rounded text-center">
              <div class="text-muted small">Last Visited</div>
              <div class="fw-semibold">{{ stats.lastVisited || '—' }}</div>
            </div>
          </div>
        </div>

        <h6 class="mb-2 text-muted">All MFE Visit Counts</h6>
        <div class="d-flex flex-wrap gap-2">
          <span
            v-for="(count, key) in stats.allVisitCounts"
            :key="key"
            class="badge bg-secondary"
          >
            {{ key }}: {{ count }}
          </span>
          <span v-if="!Object.keys(stats.allVisitCounts).length" class="text-muted small">
            No data yet — navigate to each MFE first.
          </span>
        </div>
      </div>
    </div>

    <div class="card">
      <div class="card-body">
        <h5 class="card-title">Vue MFE Details</h5>
        <ul>
          <li>Framework: Vue 3 (Composition API)</li>
          <li>Bundler: Webpack 5</li>
          <li>Integration: Web Component wrapper</li>
          <li>CSS: Bootstrap 5 (shared)</li>
        </ul>
      </div>
    </div>
  </div>
</template>

<script>
import { inject, computed } from "vue";

export default {
  name: "VueApp",
  setup() {
    const mfeStats = inject("mfeStats", null);
    const stats = computed(() =>
      mfeStats?.value ?? {
        visitCount: 0,
        allVisitCounts: {},
        userName: "",
        sessionStartTime: "",
        lastVisited: "",
      }
    );
    return { stats };
  },
};
</script>
